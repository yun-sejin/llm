Step 2: Update the DAG to Send Email Alerts on Task Failure
Modify your DAG definition to include email alerts on task failure. You can use the email_on_failure parameter and the email parameter in the task definition.

Updated sample.py

from airflow.utils.email import send_email  # Import send_email
import pendulum
from airflow.decorators import dag, task

def failure_callback(context):
    subject = f"DAG {context['dag'].dag_id} - Task {context['task_instance'].task_id} Failed"
    html_content = f"""
    <h3>DAG: {context['dag'].dag_id}</h3>
    <p>Task: {context['task_instance'].task_id} failed.</p>
    <p>Execution Date: {context['execution_date']}</p>
    <p>Log URL: <a href="{context['task_instance'].log_url}">Log</a></p>
    """
    send_email(['sj.yun@hotmail.com', 'luemier@hotmail.com'], subject, html_content)

@dag(
    schedule="*/10 * * * *",
    start_date=pendulum.datetime(2021, 1, 1, tz="UTC"),
    catchup=False,
)
def my_dag():
    @task(on_failure_callback=failure_callback)
    def first_task():
        print("This is the first task.")
        # Simulate task failure for demonstration purposes
        raise ValueError("This is a simulated task failure in first_task.")

    @task(on_failure_callback=failure_callback)
    def end_task():
        # Simulate task failure for demonstration purposes
        raise ValueError("This is a simulated task failure in end_task.")

    first_task() >> end_task()

dag_instance = my_dag()
