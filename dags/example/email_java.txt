1. Knox Mail API Java 코드 작성

// KnoxMailSender.java
import org.apache.knox.gateway.shell.KnoxSession;
import org.apache.knox.gateway.shell.KnoxSession.KnoxSessionBuilder;
import org.apache.knox.gateway.shell.email.Email;

public class KnoxMailSender {
    private String knoxUrl;
    private String username;
    private String password;

    public KnoxMailSender(String knoxUrl, String username, String password) {
        this.knoxUrl = knoxUrl;
        this.username = username;
        this.password = password;
    }

    public void sendEmail(String to, String subject, String body) throws Exception {
        KnoxSession session = KnoxSession.builder()
            .knoxUrl(knoxUrl)
            .basicAuth(username, password)
            .build();

        Email.send(session)
            .to(to)
            .subject(subject)
            .body(body)
            .execute();
    }
}

3. Python 코드에서 Java 코드 호출
Python 코드에서 Py4J를 사용하여 Java 클래스를 호출합니다.

import os
import logging
from py4j.java_gateway import JavaGateway, GatewayParameters, launch_gateway

def setup_logger(name: str) -> logging.Logger:
    logger = logging.getLogger(name)
    logger.setLevel(logging.DEBUG)
    console_handler = logging.StreamHandler()
    console_handler.setLevel(logging.DEBUG)
    formatter = logging.Formatter('%(asctime)s - %(name)s - %(levelname)s - %(message)s')
    console_handler.setFormatter(formatter)
    logger.addHandler(console_handler)
    return logger

logger = setup_logger(__name__)

def send_email_via_knox(to_email, subject, body):
    try:
        # KnoxMailSender Java 클래스를 호출하기 위해 Py4J 게이트웨이 시작
        gateway = launch_gateway(classpath="path/to/knox-mail-sender.jar")
        knox_mail_sender = gateway.jvm.KnoxMailSender(
            "https://knox.example.com:8443/gateway", "username", "password"
        )
        knox_mail_sender.sendEmail(to_email, subject, body)
        logger.info(f"Email sent to {to_email}")
    except Exception as e:
        logger.error(f"Error sending email: {e}")

# 예제 사용
send_email_via_knox("recipient@example.com", "Test Subject", "Test Body")
